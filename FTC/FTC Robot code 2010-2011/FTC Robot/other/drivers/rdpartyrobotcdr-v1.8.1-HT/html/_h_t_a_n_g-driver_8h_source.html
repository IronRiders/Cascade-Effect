00074 /**
00075  * Return the current angle
00076  * @param link the HTANG port number
00077  * @return current angle or -1 if an error occurred.
00078  */
00079 int HTANGreadAngle(tSensors link) {
00080   memset(HTANG_I2CRequest, 0, sizeof(tByteArray));
00081 
00082   HTANG_I2CRequest.arr[0] = 2;                         // Message size
00083   HTANG_I2CRequest.arr[1] = HTANG_I2C_ADDR;            // I2C Address
00084   HTANG_I2CRequest.arr[2] = HTANG_OFFSET + HTANG_ANG2; // Start Current angle
00085 
00086   if (!writeI2C(link, HTANG_I2CRequest, 2))
00087     return -1;
00088 
00089   if (!readI2C(link, HTANG_I2CReply, 2))
00090     return -1;
00091 
00092   return HTANG_I2CReply.arr[0] * 2 + HTANG_I2CReply.arr[1];
00093 }
